toSeg函式
  void toSeg(int value, int pos)
  {
    digitalWrite(D15, LOW);
    shiftOut(D2, D14, MSBFIRST, SEGMENT_MAP[value]);
    shiftOut(D2, D14, MSBFIRST, SEGMENT_SELECT[pos]);
    digitalWrite(D15, HIGH);
  }
